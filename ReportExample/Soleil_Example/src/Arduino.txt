#include <LiquidCrystal.h>

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

String inputString = "";  // 用于存储接收到的串口数据
// 自定义字符的简化版本“沈”字，分别表示字形的两部分
uint8_t shen_part1[8] = {
  B01000,
  B00100,
  B00010,
  B11100,
  B00010,
  B00100,
  B01000,
  B10000
};
uint8_t shen_part2[8] ={
  B00100,
  B11111,
  B10101,
  B00100,
  B00100,
  B00100,
  B01010,
  B10001
};
uint8_t shen_part3[] = {
  B00000,
  B11110,
  B10010,
  B10000,
  B01100,
  B01101,
  B01101,
  B00111
};

void setup() {
  lcd.begin(16, 2);  // 初始化 LCD，16 列 2 行

  // 清空 LCD 屏幕
  lcd.clear();

  // 创建两个自定义字符
  lcd.createChar(0, shen_part1);  // 第一个部分
  lcd.createChar(1, shen_part2);  // 第二个部分
  lcd.createChar(2, shen_part3);

  // 将自定义字符显示在 LCD 屏幕上
  lcd.setCursor(0, 0);  // 第一行
  lcd.write((uint8_t)0);  // 显示“沈”字的第一部分
  lcd.setCursor(1, 0);  // 第一行，第二个位置
  lcd.write((uint8_t)1);  // 显示“沈”字的第二部分
  Serial.begin(9600);  // 初始化串口通讯

}

void loop() {

  // 检查是否有数据从串口发送到 Arduino
  if (Serial.available()) {
    // 从串口读取数据并缓存到 inputString
    while (Serial.available()) {
      char incomingByte = (char)Serial.read();  // 读取一个字节
      inputString += incomingByte;  // 将字节加入到字符串中
      delay(10);
    }

    // 清空 LCD 屏幕
    lcd.clear();

    // 显示输入字符串并实现自动换行
    if (inputString.length() > 16) {
      // 如果字符数超过16个，拆分为两行显示
      lcd.print(inputString.substring(0, 16));  // 显示第一行
      lcd.setCursor(0, 1);  // 将光标移到第二行
      lcd.print(inputString.substring(16));  // 显示第二行
    } else {
      lcd.print(inputString);  // 否则直接显示
    }

    // 输出到串口以供调试
    Serial.println(inputString);

    // 清空 inputString
    inputString = "";
  }
}
